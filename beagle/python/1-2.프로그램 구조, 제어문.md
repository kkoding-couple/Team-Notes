# 프로그램 구조, 제어문

## 조건문

> "돈이 있으면 택시를 타고, 돈이 없으면 걸어 간다."

* if문

```python
# 기본 구조
if 조건문:
  수행할 문장1
  수행할 문장2
  ...
else:
  수행할 문장A
  수행할 문장B
  ...
  
# 들여쓰기는 Spacebar(공백) 4개 사용을 권장, Tab 1개도 가능

money = True
if money:
  print('택시를 타고 가라')
else:
  print('걸어 가라')

money = 2000
if money >= 3000:
  print('택시를 타고 가라')
else:
  print('걸어 가라')
  
# 논리 연산자: and, or, not, in
money = 2000
card = True
if money >= 3000 or card:
  print('택시를 타고 가라')
else:
  print('걸어 가라')
  
# 다양한 조건 판단
pocket = ['note', 'phone']
friend = 2
if 'money' in pocket:
  print('택시를 타고 가라')
elif friend > 0:
  print('택시를 타고 가라')
else:
  print('걸어 가라')
```

## 반복문

> 반복해서 문장을 수행해야 할 경우

* while문

```python
# 기본 구조
while 조건문:
  수행할 문장1
  수행할 문장2
  ...
  
tree_hit = 0
while tree_hit < 10:
  tree_hit = tree_hit + 1  # tree_hit += 1
  print('나무를 {}번 찍었습니다.'.format(tree_hit))
if tree_hit == 10:
	print('나무 넘어갑니다.')
  
# while문 강제로 빠져나가기
coffee = 10
while True:
  coffee = coffee - 1  # coffee -= 1
  print('남은 커피의 양은 {}개 입니다.'.format(coffee))
  if coffee == 0:
    print('커피가 다 떨어졌습니다. 판매를 중지합니다.')
    break

coffee = 10
while True:
  money = int(input('돈을 넣어 주세요: '))
  if money == 500:
    coffee = coffee - 1  # coffee -= 1
    print('커피를 줍니다.')
  elif money > 500:
    coffee = coffee - 1  # coffee -= 1
    print('거스름돈 {}를 주고 커피를 줍니다.'.format(money - 500))
  else:
    print('돈을 다시 돌려주고 커피를 주지 않습니다.')
  if coffee == 0:
    print('커피가 다 떨어졌습니다. 판매를 중지 합니다.')
    break
    
# 무한 루프
while True:
  print("Ctrl + c를 눌러야 whild문을 빠져나갈 수 있습니다.")
```
